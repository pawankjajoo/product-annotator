// ═══════════════════════════════════════════════════
//  IMAGE UPLOAD HANDLING
//  Supports both click-to-upload and drag & drop.
//  We use FileReader to get a base64 data URL which
//  lets us display it immediately without a server.
// ═══════════════════════════════════════════════════

function handleImageUpload(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  // Once the file is read, update the canvas to show the uploaded image
  reader.onload = (ev) => {
    mainImageSrc = ev.target.result;
    mainImageDataUrl = ev.target.result;
    // Clear hosted URL when a new file is uploaded locally — avoids conflicts
    mainImageHostedUrl = '';
    const urlInput = document.getElementById('main-image-url');
    if (urlInput) urlInput.value = '';
    const urlStatus = document.getElementById('main-url-status');
    if (urlStatus) urlStatus.textContent = '';
    document.getElementById('main-image').src = mainImageSrc;
    document.getElementById('upload-zone').style.display = 'none';
    document.getElementById('canvas-wrapper').style.display = 'inline-block';
  };
  reader.readAsDataURL(file);
}

// Set up drag and drop on the canvas area so user can just toss an image in
const canvasArea = document.getElementById('canvas-area');
canvasArea.addEventListener('dragover', (e) => { e.preventDefault(); e.stopPropagation(); });
canvasArea.addEventListener('drop', (e) => {
  e.preventDefault();
  e.stopPropagation();
  const file = e.dataTransfer.files[0];
  // Only accept image files — we don't want someone dropping a PDF in here
  if (file && file.type.startsWith('image/')) {
    const reader = new FileReader();
    reader.onload = (ev) => {
      mainImageSrc = ev.target.result;
      mainImageDataUrl = ev.target.result;
      mainImageHostedUrl = '';
      document.getElementById('main-image').src = mainImageSrc;
      document.getElementById('upload-zone').style.display = 'none';
      document.getElementById('canvas-wrapper').style.display = 'inline-block';
    };
    reader.readAsDataURL(file);
  }
});

// Live feedback as user types the main image URL
function onMainImageUrlChange(val) {
  const status = document.getElementById('main-url-status');
  if (!val.trim()) { status.textContent = ''; return; }
  if (val.startsWith('http://') || val.startsWith('https://')) {
    status.textContent = '✓ URL looks valid — click Apply to use it';
    status.style.color = '#4caf50';
  } else {
    status.textContent = 'URL should start with https://';
    status.style.color = '#e57373';
  }
}

// Apply the hosted URL as the main product image
function applyMainImageUrl() {
  const urlInput = document.getElementById('main-image-url');
  const url = urlInput.value.trim();
  if (!url) return;
  const status = document.getElementById('main-url-status');
  // Set the hosted url — this will be preferred in embed code generation
  mainImageHostedUrl = url;
  mainImageSrc = url;
  document.getElementById('main-image').src = url;
  document.getElementById('upload-zone').style.display = 'none';
  document.getElementById('canvas-wrapper').style.display = 'inline-block';
  status.textContent = '✓ Hosted URL applied — embed code will use this URL';
  status.style.color = '#4caf50';
}

// Save a hosted URL for an annotation's media image
function saveMediaUrl(val) {
  const ann = annotations.find(a => a.id === selectedId);
  if (!ann) return;
  const status = document.getElementById('media-url-status');
  if (!val.trim()) {
    ann.mediaImageUrl = '';
    status.textContent = '';
    return;
  }
  if (val.startsWith('http://') || val.startsWith('https://')) {
    ann.mediaImageUrl = val.trim();
    // If no local upload exists, also set mediaImage to the URL for preview
    if (!ann.mediaImage || ann.mediaImage.startsWith('http')) {
      ann.mediaImage = val.trim();
    }
    status.textContent = '✓ Embed code will use this hosted URL';
    status.style.color = '#4caf50';
    renderAll();
  } else {
    status.textContent = 'URL should start with https://';
    status.style.color = '#e57373';
  }
}

//  Create, read, update, delete annotation objects.
//  Each annotation has: id, subtitle, title,
//  description, mediaImage, top (%), left (%)
// ═══════════════════════════════════════════════════

function addAnnotation() {
  // Generate a unique id using timestamp + random to avoid colisions
  const id = Date.now() + Math.random();
  annotations.push({
    id,
    subtitle: '',
    title: 'New Annotation',
    description: '',
    mediaImage: '',
    mediaImageUrl: '',
    top: 50,
    left: 50
  });
  // Auto-select the new annotation so user can start editing right away
  selectAnnotation(id);
  renderAll();
}

function removeAnnotation(id) {
  annotations = annotations.filter(a => a.id !== id);
  // If the deleted annotation was selected, clear the editor panel
  if (selectedId === id) { selectedId = null; showEditorEmpty(); }
  renderAll();
}

function selectAnnotation(id) {
  selectedId = id;
  const ann = annotations.find(a => a.id === id);
  if (!ann) return;

  // Show the editor form and hide the empty state placeholder
  document.getElementById('editor-empty').style.display = 'none';
  document.getElementById('editor-form').style.display = 'flex';

  // Populate all the form fields with the selected annotation's data
  document.getElementById('ed-subtitle').value = ann.subtitle;
  document.getElementById('ed-title').value = ann.title;
  document.getElementById('ed-desc').value = ann.description;
  document.getElementById('ed-top').value = parseFloat(ann.top);
  document.getElementById('ed-left').value = parseFloat(ann.left);

  // Handle the media image preview — show thumbnail if one exists
  const mp = document.getElementById('media-preview');
  const mt = document.getElementById('media-upload-text');
  if (ann.mediaImage) {
    mp.innerHTML = '<img src="'+ann.mediaImage+'" style="max-width:100%;max-height:120px;border-radius:4px;margin-bottom:8px;"><br><span class="remove-img" onclick="event.stopPropagation();removeMediaImage()">Remove image</span>';
    mt.style.display = 'none';
  } else {
    mp.innerHTML = '';
    mt.style.display = 'block';
  }

  // Populate the hosted media URL field if one was previously set
  const mediaUrlInput = document.getElementById('ed-media-url');
  const mediaUrlStatus = document.getElementById('media-url-status');
  if (mediaUrlInput) {
    mediaUrlInput.value = ann.mediaImageUrl || '';
    if (ann.mediaImageUrl) {
      mediaUrlStatus.textContent = '✓ Embed code will use this hosted URL';
      mediaUrlStatus.style.color = '#4caf50';
    } else {
      mediaUrlStatus.textContent = '';
    }
  }
  renderAll();
}

      selectAnnotation(selectedId);

  reader.readAsDataURL(file);
}

// Render the emoji picker grid with all categories and their emojis
function renderEmojiGrid(filter = '') {
  const grid = document.getElementById('emoji-grid');
  let html = '';
  const filterLower = filter.toLowerCase();

  if (filter) {
    // When filtering, just show a flat list of matching emojis
    // (we cant really search by emoji name so we just filter by the charcter itself,
    // but its still useful for quick scrolling through results)
    const filtered = ALL_EMOJIS.filter(e => e.includes(filter));
    if (filtered.length === 0) {
      html = '<div style="grid-column:1/-1;text-align:center;color:#666;font-size:12px;padding:16px;">No matching emojis found</div>';
    } else {
      filtered.forEach(e => {
        html += '<button onclick="selectEmoji(\'' + e + '\')" ' + (e === iconEmoji ? 'class="selected"' : '') + '>' + e + '</button>';
      });
    }
  } else {
    // No filter — show emojis organized by category with headers
    EMOJI_CATEGORIES.forEach(cat => {
      html += '<div class="emoji-category-label">' + cat.name + '</div>';
      cat.emojis.forEach(e => {
        html += '<button onclick="selectEmoji(\'' + e + '\')" ' + (e === iconEmoji ? 'class="selected"' : '') + '>' + e + '</button>';
      });
    });
  }
  grid.innerHTML = html;
}

// Rebuild the hotspot buttons overlaid on the canvas image.
// This gets called frequnetly during drag operations so it needs to be reasonably fast.
function renderHotspots() {
  const container = document.getElementById('hotspot-container');
  if (!container) return;

  // Determine what goes inside each button — either the plus SVG or the chosen emoji
  const iconContent = iconMode === 'plus'
    ? PLUS_SVG
    : '<span class="emoji-icon">' + iconEmoji + '</span>';

      onclick="selectAnnotation(${ann.id})">
      ${iconContent}

  `).join('');
}


// ═══════════════════════════════════════════════════
//  DRAG & DROP FOR HOTSPOT REPOSITONING
//  When the user mousedowns on a hotspot button we
//  track mouse movement and update the annotation's
//  top/left percentages in realtime. Releasing the
//  mouse finishes the drag opreation.
// ═══════════════════════════════════════════════════

function startDrag(e, id) {
  e.preventDefault();
  const wrapper = document.getElementById('canvas-wrapper');
  const rect = wrapper.getBoundingClientRect();
  dragState = { id, rect };
  const btn = e.target.closest('.hotspot-btn');
  if (btn) btn.classList.add('dragging');
  selectAnnotation(id);

  // Mouse move handler — recalculate position as percentage of wrapper dimensions
  function onMove(ev) {
    const x = ((ev.clientX - rect.left) / rect.width * 100);
    const y = ((ev.clientY - rect.top) / rect.height * 100);
    const ann = annotations.find(a => a.id === id);
    if (ann) {
      // Clamp values between 0 and 100 so the button cant leave the image boundaries
      ann.left = Math.max(0, Math.min(100, x)).toFixed(1);
      ann.top = Math.max(0, Math.min(100, y)).toFixed(1);
      renderHotspots();
      // Also update the number inputs in the editor if this annotation is selected
      if (selectedId === id) {
        document.getElementById('ed-top').value = ann.top;
        document.getElementById('ed-left').value = ann.left;
      }
      renderSidebar();
    }
  }

  // Mouse up handler — clean up the drag state and remove event listners
  function onUp() {
    document.querySelectorAll('.hotspot-btn').forEach(b => b.classList.remove('dragging'));
    dragState = null;
    document.removeEventListener('mousemove', onMove);
    document.removeEventListener('mouseup', onUp);
  }

  document.addEventListener('mousemove', onMove);
  document.addEventListener('mouseup', onUp);
}

//  displays it in a modal overlay. This uses the same
//  markup structure as the embed code ouput so what

function showPreview() {
  // Make sure the user has actually set something up before previewing
  if (!mainImageSrc || annotations.length === 0) {
    alert('Upload an image and add at least one annotation first.');
    return;
  }
  const html = generatePreviewHTML();
  document.getElementById('preview-content').innerHTML = html;
  document.getElementById('preview-modal').classList.add('show');

  // Wire up the click handlers on the preview hotspot buttons
  // We use setTimeout because the HTML needs to be in the DOM first
  setTimeout(() => {
    document.querySelectorAll('.hp-btn').forEach(btn => {
      btn.addEventListener('click', function() {
        const idx = this.getAttribute('data-index');
        // Hide all content blocks and deactivate all buttons
        document.querySelectorAll('.hp-content-block').forEach(b => b.classList.remove('hp-visible'));
        document.querySelectorAll('.hp-btn').forEach(b => b.classList.remove('hp-active'));
        // Then show the clicked one
        this.classList.add('hp-active');
        const target = document.querySelector('.hp-content-block[data-index="'+idx+'"]');
        if (target) target.classList.add('hp-visible');
      });
    });
  }, 100);
}

// Build the preview HTML — basically the same as what the embed code will produce
function generatePreviewHTML() {
  // Generate the icon markup based on current mode selection
  const iconMarkup = iconMode === 'plus'
    ? PLUS_SVG
    : '<span class="emoji-icon">' + iconEmoji + '</span>';

  let btnsHtml = annotations.map((ann, i) => `
    <button class="hp-btn ${i===0?'hp-active':''}" data-index="${i}" style="top:${ann.top}%;left:${ann.left}%;background:${buttonColor};">
      ${iconMarkup}
    </button>
  `).join('');

  let blocksHtml = annotations.map((ann, i) => `
    <div class="hp-content-block ${i===0?'hp-visible':''}" data-index="${i}">
      ${ann.mediaImage ? '<img src="'+ann.mediaImage+'" alt="">' : ''}
      ${ann.subtitle ? '<h4>'+ann.subtitle+'</h4>' : ''}
      <h3>${ann.title}</h3>
      <p>${ann.description.replace(/\n/g, '<br>')}</p>
    </div>
  `).join('');

  return `
    <div class="hotspots-preview-wrapper">
      <div class="hotspots-preview-image-area">
        <img src="${mainImageSrc}" alt="">
        ${btnsHtml}
      </div>
      <div class="hotspots-preview-content">${blocksHtml}</div>
    </div>
  `;
}

// Escape HTML special characters to prevent injection in the ouput code
function escHtml(str) {
  const div = document.createElement('div');
  div.textContent = str;
  return div.innerHTML;
}

// Copy the generated embed code to clipbaord with visual feedback
function copyCode() {
  const code = document.getElementById('code-output').textContent;
  navigator.clipboard.writeText(code).then(() => {
    const fb = document.getElementById('copy-feedback');
    fb.classList.add('show');
    setTimeout(() => fb.classList.remove('show'), 2000);
  });
}

// ═══════════════════════════════════════════════════
//  PROJECT PERSISTENCE (localStorage)
//  Auto-saves the entire project state every 3 sec
//  so the user doesn't lose thier work if they
//  accidentally close the tab or refresh the page.
//  On load we check for exisiting saved data and
//  restore everything to where they left off.
// ═══════════════════════════════════════════════════

function saveProject() {
  const data = {
    annotations,
    mainImageSrc: mainImageDataUrl,
    mainImageHostedUrl,  // persist the hosted URL separately
    buttonColor,
    iconMode,     // persist the icon mode selecton
    iconEmoji     // and which emoji was chosen
  };
  try { localStorage.setItem('hotspot-editor-project', JSON.stringify(data)); } catch(e) {}
}

function loadProject() {
  try {
    const raw = localStorage.getItem('hotspot-editor-project');
    if (!raw) return;
    const data = JSON.parse(raw);

    // Restore annotation data
    if (data.annotations) annotations = data.annotations;

    // Restore the main product image if one was prevously uploaded
    if (data.mainImageSrc) {
      mainImageSrc = data.mainImageSrc;
      mainImageDataUrl = data.mainImageSrc;
      document.getElementById('main-image').src = mainImageSrc;
      document.getElementById('upload-zone').style.display = 'none';
      document.getElementById('canvas-wrapper').style.display = 'inline-block';
    }

    // Restore hosted URL if one was set
    if (data.mainImageHostedUrl) {
      mainImageHostedUrl = data.mainImageHostedUrl;
      const urlInput = document.getElementById('main-image-url');
      if (urlInput) urlInput.value = mainImageHostedUrl;
    }

    // Restore button color
    if (data.buttonColor) {
      buttonColor = data.buttonColor;
      document.getElementById('btn-color').value = buttonColor;
      document.getElementById('btn-color-hex').textContent = buttonColor;
    }

    // Restore icon mode and emoji selection
    if (data.iconMode) {
      iconMode = data.iconMode;
      setIconMode(iconMode); // this also updates the UI toggle btns
    }
    if (data.iconEmoji) {
      iconEmoji = data.iconEmoji;
    }

    renderAll();
    updateIconPreview();
  } catch(e) {
    // If anything goes wrong with loading, just start fresh — don't crash the app
  }
}

// Save every 3 seconds — frequent enought to not lose much work but
// not so frequent that it hammers localStorage on every keystroke
setInterval(saveProject, 3000);

//  and populate the emoji picker grid so its ready
//  to go when the user switches to emoji mode.
// ═══════════════════════════════════════════════════

loadProject();
renderAll();
renderEmojiGrid();
